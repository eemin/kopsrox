multipass exec k3s-m -- \
  bash -c "curl -sfL https://get.k3s.io | sh -"
TOKEN=$(multipass exec k3s-m sudo cat /var/lib/rancher/k3s/server/node-token)
IP=$(multipass info k3s-m | grep IPv4 | awk '{print $2}')
multipass exec k3s-w1 -- \
bash -c "curl -sfL https://get.k3s.io | K3S_URL=\"https://$IP:6443\" K3S_TOKEN=\"$TOKEN\" sh -"
multipass exec k3s-w2 -- \
bash -c "curl -sfL https://get.k3s.io | K3S_URL=\"https://$IP:6443\" K3S_TOKEN=\"$TOKEN\" sh -"
sleep 30
multipass exec k3s-m -- sudo kubectl get nodes
